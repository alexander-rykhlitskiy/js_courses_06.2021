```js
function getMax(numbers) {
  let max = numbers[0];
  for (let i = 0; i < numbers.length; i++) {
    if (numbers[i] > max) {
      max = numbers[i];
    }
  }
```

```js
  let ul = document.querySelector('ul');

    li.textContent = car.name + ' - ' + car.age;

  }
```
